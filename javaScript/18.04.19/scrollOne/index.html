Scroll & Scroll Scroll & Scroll

X & Y

The screenX and screenY properties returns the x (horizontal) and y (vertical) coordinates of the window relative to the screen.

Scroll Event

                window.addEventListener('scroll', () => {
                    console.log('Scrolled!');
                });

Scroll Vertical

                window.addEventListener('scroll', () => {
                    /* console.log('Scrolled!'); */
                    const yValue = window.scrollY;
                    console.log(yValue); 
                });

Scroll Height Value

                window.addEventListener('scroll', () => {
                    /* console.log('Scrolled!'); */
                    const yValue = window.scrollY;
                    /* console.log(yValue); */
                    const height = document.documentElement.scrollHeight;
                    console.log(height);
                });

Scroll Bottom

                window.addEventListener('scroll', () => {
                    /* console.log('Scrolled!'); */
                    const yValue = window.scrollY;
                    /* console.log(yValue); */
                    /* const height = document.documentElement.scrollHeight;
                    console.log(height); */
                   const bottom = document.documentElement.scrollHeight - window.innerHeight;
                   /* console.log(bottom); */
        
                   if(Math.ceil(yValue) === bottom){
                       alert('You reached BOTTOM ! VALUE = 1077 !');
                   }
                });
